void setup()
{
  Serial.begin(9600);
}

void loop()  
{
  float normH = 100*(analogRead(A0)-432) /(800-432);
  float normH1 = 100*(analogRead(A1)-541) /(852-541);
  float normH2 = 100*(analogRead(A2)-539) /(857-539);
  float normH3 = 100*(analogRead(A3)-491) /(820-491);

Serial.print(normH);
  Serial.print(",");
  Serial.print(normH1);
  Serial.print(",");
  Serial.print(normH2);
  Serial.print(",");
  Serial.println(normH3);
delay(200);
}
